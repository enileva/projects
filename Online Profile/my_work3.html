Fortune Cookie


import random print("Welcome to the Fortune Cookie program.") input("Press enter to receive your fortune. ") fortuneNumber = random.randint(1,5) if fortuneNumber == 1: print("A beautiful, smart, and loving person will be coming into your life.") elif fortuneNumber == 2: print("Chance favors those in motion.") elif fortuneNumber == 3: print("In order to take, one must first give.") elif fortuneNumber == 4: print("Like the river flows into the sea. Some things are just meant to be.") elif fortuneNumber == 5: print("The best prediction of future is the past.")
